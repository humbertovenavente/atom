```typescript
import mongoose from 'mongoose';

const nodeConfigSchema = new mongoose.Schema({
  systemPrompt: { type: String, default: '' },
  temperature: { type: Number, default: 0.7 },
}, { _id: false });

const flowNodeSchema = new mongoose.Schema({
  id: { type: String, required: true },
  type: { type: String, required: true },
  position: {
    x: { type: Number, required: true },
    y: { type: Number, required: true },
  },
  data: {
    label: String,
    icon: String,
    color: String,
    config: { type: nodeConfigSchema, default: undefined },
  },
}, { _id: false });

const flowEdgeSchema = new mongoose.Schema({
  id: String,
  source: String,
  target: String,
  animated: Boolean,
}, { _id: false });

const flowSchema = new mongoose.Schema({
  flowId: { type: String, unique: true, required: true },
  nodes: [flowNodeSchema],
  edges: [flowEdgeSchema],
  nodeConfigs: { type: mongoose.Schema.Types.Mixed, default: {} },
}, { timestamps: true });

export const Flow = mongoose.models['Flow'] || mongoose.model('Flow', flowSchema);
```

**POST /api/flow (src/server/routes/api/flow.post.ts):** Create new file. Pattern: `defineEventHandler` + `await connectDB()` + `readBody` + `Flow.findOneAndUpdate` with upsert. Always set `flowId: 'default'` regardless of what the client sends.

```typescript
import { defineEventHandler, readBody } from 'h3';
import { connectDB } from '../../db/connect';
import { Flow } from '../../models/flow';

export default defineEventHandler(async (event) => {
  await connectDB();
  const body = await readBody(event);
  await Flow.findOneAndUpdate(
    { flowId: 'default' },
    { ...body, flowId: 'default' },
    { upsert: true, new: true }
  );
  return { ok: true };
});
```

**Modified GET /api/flow (src/server/routes/api/flow.get.ts):** Modify the existing file. Keep the hardcoded `nodes` and `edges` arrays as-is (they are the default). Add MongoDB lookup BEFORE returning defaults. Support `?default=true` query param to skip MongoDB and return hardcoded defaults (used by resetFlow).

Per RESEARCH.md Pitfall 2: ALWAYS `await connectDB()` first. Wrap the MongoDB lookup in try/catch so DB unavailability falls through to hardcoded defaults gracefully.

```typescript
import { defineEventHandler, getQuery } from 'h3';
import { connectDB } from '../../db/connect';
import { Flow } from '../../models/flow';
import type { FlowNode, FlowEdge } from '../../../shared/types';

// ... existing hardcoded nodes and edges arrays stay here unchanged ...

export default defineEventHandler(async (event) => {
  const query = getQuery(event);

  // Skip MongoDB lookup when reset is requested
  if (query.default !== 'true') {
    try {
      await connectDB();
      const saved = await Flow.findOne({ flowId: 'default' }).lean();
      if (saved) {
        return {
          nodes: saved.nodes,
          edges: saved.edges,
          nodeConfigs: saved.nodeConfigs ?? {},
        };
      }
    } catch {
      // DB unavailable — fall through to hardcoded default
    }
  }

  return { nodes, edges };
});
```

Note: Import `getQuery` from 'h3' to read the query parameter. The handler changes from sync to async.

```typescript
import { Component, inject } from '@angular/core';
import { FlowService } from '../../services/flow.service';

@Component({
  selector: 'app-flow-toolbar',
  standalone: true,
  imports: [],
  template: `
    <div class="bg-gray-900 border-b border-gray-700 px-4 py-2 flex items-center gap-3">
      <button (click)="save()"
        [disabled]="flowService.saveStatus() === 'saving'"
        class="text-sm border border-gray-600 text-gray-300 hover:text-white hover:border-gray-400 px-3 py-1.5 rounded-lg transition-colors disabled:opacity-50">
        @switch (flowService.saveStatus()) {
          @case ('saving') { Guardando... }
          @case ('saved') { Guardado }
          @case ('error') { Error — Reintentar }
          @default { Guardar }
        }
      </button>
      <button (click)="reset()"
        class="text-sm border border-gray-600 text-gray-300 hover:text-red-400 hover:border-red-400 px-3 py-1.5 rounded-lg transition-colors">
        Reset Flow
      </button>
    </div>
  `,
})
export class FlowToolbarComponent {
  readonly flowService = inject(FlowService);

  save(): void {
    this.flowService.saveFlow();
  }

  reset(): void {
    if (window.confirm('¿Restablecer el flujo al default?')) {
      this.flowService.resetFlow();
    }
  }
}
```
